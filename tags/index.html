async function sendLogAI() { const el = document.getElementById('logai-input'); const msg = el.value.trim(); if (!msg) return; el.value = ''; addLogMsg(msg, false); try { const res = await fetch(`${API_URL}/send`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ sessionId: logSid, message: msg, clientId: 'client_arhitektor' // Твой ID }) }); const data = await res.json(); if (!data.messageId) throw new Error('No ID'); const aiMsg = addLogMsg('Печатает...', true); let startTime = Date.now(); let interval = 3000; let errorCount = 0; // Счётчик ошибок const busyText = 'Я всё еще обрабатываю ваш запрос...'; const poll = async () => { const elapsed = (Date.now() - startTime) / 1000; // Останавливаем через 15 минут if (elapsed > 900) { aiMsg.innerText = 'Извините, я призадумался. Попробуйте заглянуть позже.'; return; } try { const r = await fetch(`${API_URL}/get-answer/${data.messageId}`); const d = await r.json(); if (d.answer) { aiMsg.innerText = d.answer; // Ответ пришел! Завершаем поллинг return; } // Обновляем статус, если ответа нет if (elapsed > 60 && elapsed < 180) aiMsg.innerText = 'Вопрос сложный, думаю...'; if (elapsed > 180 && aiMsg.innerText !== busyText) { aiMsg.innerText = busyText; // Обновляем только если текст не тот же interval = 30000; // После 3-х минут увеличиваем интервал } errorCount = 0; // Сброс ошибок при успешном поллинге setTimeout(poll, interval); } catch (e) { console.error("Ошибка поллинга", e); errorCount++; // Если ошибка сервера больше 3 раз подряд — прекращаем поллинг if (errorCount >= 3) { aiMsg.innerText = 'Проблемы на сервере. Попробуйте позже.'; return; } // Если ошибка не критична, продолжаем попытки setTimeout(poll, interval); } }; poll(); // Запуск поллинга } catch (e) { addLogMsg('Ошибка связи. Попробуйте позже.', true); } }
